using System;
using System.Collections;
using System.Collections.Generic;
using System.Linq;
using System.Security.Cryptography;
using System.Security.Cryptography.X509Certificates;
using System.Text;
using System.Threading;
using System.Threading.Tasks;

namespace cours23
{
    internal class Program
    {



        static void Main()
        {
            string expression = "hello world hello universe";


            Dictionary<string, int> translate = new Dictionary<string, int>();
           

            foreach(var word in expression.Split(' '))
            {
                if(!translate.ContainsKey(word))
                    translate.Add(word, 1);
                else
                    translate[word]++;
            }

            foreach (var word in translate)
            {
                Console.WriteLine($"{word.Key}: {word.Value}");
            }

            Console.ReadKey();

        }
    }
}
